n = int(input())

best_value = -1
best_weight = 0
best_time = 0
best_quality = 0

for i in range(n):
    weight = int(input())
    time = int(input())
    quality = int(input())

    value = (weight / time) ** quality

    if value > best_value:
        best_value = int(value)
        best_weight = int(weight)
        best_time = int(time)
        best_quality = int(quality)

print(f"{best_weight} : {best_time} = {best_value} ({best_quality})")
